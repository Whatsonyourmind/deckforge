Integrate in Minutes

TypeScript SDK, Python, or plain curl. Pick your stack.

app.ts 
import { DeckForge } from '@deckforge/sdk';

const df = new DeckForge({ apiKey: process.env.DECKFORGE_API_KEY });

// Generate from natural language
const deck = await df.generate({
  prompt: 'Q3 2026 earnings for ACME Corp',
  slideCount: 12,
  theme: 'executive_dark',
});

// Or build with the typed builder
const ir = df.presentation('Q3 Earnings')
  .addSlide(df.slides.title({ title: 'ACME Corp Q3 2026' }))
  .addSlide(df.slides.chart({
    title: 'Revenue Trend',
    chart: df.charts.bar({
      categories: ['Q1', 'Q2', 'Q3'],
      series: [{ name: 'Revenue', values: [1.2, 1.8, 2.4] }],
    }),
  }))
  .build();

const pptx = await df.render(ir);
console.log(`Deck ready: ${pptx.downloadUrl}`);

# Generate a presentation from a prompt
curl -X POST https://api.deckforge.dev/v1/generate \
  -H "X-API-Key: $DECKFORGE_API_KEY" \
  -H "Content-Type: application/json" \
  -d '{
    "prompt": "Q3 2026 earnings for ACME Corp",
    "slide_count": 12,
    "theme": "executive_dark"
  }'

# Render an IR payload to PPTX
curl -X POST https://api.deckforge.dev/v1/render \
  -H "X-API-Key: $DECKFORGE_API_KEY" \
  -H "Content-Type: application/json" \
  -d '@presentation.json' \
  --output deck.pptx

# List available themes
curl https://api.deckforge.dev/v1/discovery/themes \
  -H "X-API-Key: $DECKFORGE_API_KEY"

# x402 payment (AI agents -- no API key needed)
curl -X POST https://api.deckforge.dev/v1/render \
  -H "Content-Type: application/json" \
  -H "X-Payment: <x402-payment-header>" \
  -d '@presentation.json'
main.py 
import httpx
import os

API_KEY = os.environ["DECKFORGE_API_KEY"]
BASE_URL = "https://api.deckforge.dev"

headers = {
    "X-API-Key": API_KEY,
    "Content-Type": "application/json",
}

# Generate from natural language
resp = httpx.post(
    f"{BASE_URL}/v1/generate",
    headers=headers,
    json={
        "prompt": "Q3 2026 earnings for ACME Corp",
        "slide_count": 12,
        "theme": "executive_dark",
    },
)
job = resp.json()
print(f"Job: {job['job_id']} -- {job['status']}")

# Poll for completion
result = httpx.get(
    f"{BASE_URL}/v1/jobs/{job['job_id']}",
    headers=headers,
)
print(f"Download: {result.json()['download_url']}")
