```bash
#!/usr/bin/env bash
set -euo pipefail

# Verify Docker Compose brings up all DeckForge services.
# Expects: docker compose up -d already running.
# Exits 0 on success, non-zero on failure.

echo "=== DeckForge Docker Compose Verification ==="

EXPECTED_SERVICES=("api" "content-worker" "render-worker" "postgres" "redis" "minio")

# Step 1: Check all services are running
echo "[1/3] Checking service status..."
for svc in "${EXPECTED_SERVICES[@]}"; do
  status=$(docker compose ps --format json "$svc" 2>/dev/null | python3 -c "import sys,json; d=json.load(sys.stdin); print(d.get('State','unknown'))" 2>/dev/null || echo "not_found")
  if [[ "$status" != "running" ]]; then
    echo "  FAIL: $svc is $status (expected running)"
    echo "  Logs:"
    docker compose logs --tail=20 "$svc" 2>/dev/null || true
    exit 1
  fi
  echo "  OK: $svc is running"
done

# Step 2: Check health endpoints
echo "[2/3] Checking service health..."

# Postgres
docker compose exec -T postgres pg_isready -U deckforge > /dev/null 2>&1 \
  && echo "  OK: PostgreSQL accepting connections" \
  || { echo "  FAIL: PostgreSQL not ready"; exit 1; }

# Redis
docker compose exec -T redis redis-cli ping > /dev/null 2>&1 \
  && echo "  OK: Redis responding" \
  || { echo "  FAIL: Redis not responding"; exit 1; }

# API health
echo "[3/3] Checking API health endpoint..."
HTTP_CODE=$(curl -s -o /dev/null -w "%{http_code}" http://localhost:8000/v1/health 2>/dev/null || echo "000")
if [[ "$HTTP_CODE" == "200" ]]; then
  echo "  OK: API health returns 200"
  curl -s http://localhost:8000/v1/health | python3 -m json.tool
else
  echo "  FAIL: API health returned $HTTP_CODE (expected 200)"
  echo "  API logs:"
  docker compose logs --tail=30 api 2>/dev/null || true
  exit 1
fi

echo ""
echo "=== All services healthy ==="
```

```bash
#!/usr/bin/env bash
set -euo pipefail

# DeckForge End-to-End Smoke Test
# Verifies the full render pipeline: API -> Worker -> .pptx output
# Requires: services running (docker compose up -d), database bootstrapped
# Exits 0 on success, non-zero on failure.

SCRIPT_DIR="$(cd "$(dirname "${BASH_SOURCE[0]}")" && pwd)"
PROJECT_DIR="$(dirname "$SCRIPT_DIR")"
OUTPUT_DIR="$PROJECT_DIR/tmp/smoke-test"
API_URL="${DECKFORGE_API_URL:-http://localhost:8000}"

mkdir -p "$OUTPUT_DIR"

echo "=== DeckForge Smoke Test ==="

# Step 1: Health check
echo "[1/4] Health check..."
HTTP_CODE=$(curl -s -o /dev/null -w "%{http_code}" "$API_URL/v1/health")
if [[ "$HTTP_CODE" != "200" ]]; then
  echo "  FAIL: Health check returned $HTTP_CODE"
  exit 1
fi
echo "  OK: API is healthy"

# Step 2: Get or create test API key
# If DECKFORGE_TEST_API_KEY is set, use it. Otherwise try to create one via bootstrap.
API_KEY="${DECKFORGE_TEST_API_KEY:-}"
if [[ -z "$API_KEY" ]]; then
  echo "[2/4] No DECKFORGE_TEST_API_KEY set, attempting to read from DB..."
  # Try to get a key from the database directly
  API_KEY=$(docker compose exec -T postgres psql -U deckforge -d deckforge -t -c \
    "SELECT prefix || '...' FROM api_keys LIMIT 1;" 2>/dev/null | tr -d ' \n' || echo "")
  if [[ -z "$API_KEY" || "$API_KEY" == "..." ]]; then
    echo "  WARN: No API key found. Run: bash scripts/bootstrap-db.sh"
    echo "  Then set: export DECKFORGE_TEST_API_KEY=dk_test_..."
    echo "  Skipping render test (health check passed)."
    exit 0
  fi
  echo "  Found key prefix: $API_KEY"
  echo "  NOTE: Cannot retrieve full key from DB. Set DECKFORGE_TEST_API_KEY for full test."
  echo "  Skipping render test (no usable key)."
  exit 0
fi

# Step 3: Render a test presentation
echo "[3/4] Testing render pipeline..."
RENDER_BODY='{
  "presentation": {
    "metadata": {"title": "Smoke Test", "author": "DeckForge CI"},
    "theme": "corporate-blue",
    "slides": [
      {
        "slide_type": "title",
        "title": "DeckForge Smoke Test",
        "subtitle": "Automated verification"
      },
      {
        "slide_type": "bullets",
        "title": "Key Features",
        "bullets": ["32 slide types", "15 themes", "AI generation", "Finance vertical"]
      },
      {
        "slide_type": "two_column",
        "title": "Two Column Layout",
        "left": {"heading": "Left Side", "bullets": ["Point A", "Point B"]},
        "right": {"heading": "Right Side", "bullets": ["Point C", "Point D"]}
      }
    ]
  }
}'

OUTPUT_FILE="$OUTPUT_DIR/smoke-test.pptx"
HTTP_CODE=$(curl -s -o "$OUTPUT_FILE" -w "%{http_code}" \
  -X POST "$API_URL/v1/render" \
  -H "Authorization: Bearer $API_KEY" \
  -H "Content-Type: application/json" \
  -d "$RENDER_BODY")

if [[ "$HTTP_CODE" != "200" ]]; then
  echo "  FAIL: Render returned $HTTP_CODE"
  cat "$OUTPUT_FILE" 2>/dev/null || true
  exit 1
fi

# Validate output
FILE_SIZE=$(stat -f%z "$OUTPUT_FILE" 2>/dev/null || stat -c%s "$OUTPUT_FILE" 2>/dev/null || echo "0")
if [[ "$FILE_SIZE" -lt 5000 ]]; then
  echo "  FAIL: Output file too small (${FILE_SIZE} bytes, expected >5000)"
  exit 1
fi

# Check PK zip header (PPTX is a ZIP file)
HEADER=$(xxd -l 2 -p "$OUTPUT_FILE" 2>/dev/null || echo "0000")
if [[ "$HEADER" != "504b" ]]; then
  echo "  FAIL: File does not have PK zip header (got: $HEADER)"
  exit 1
fi

echo "  OK: Render produced valid .pptx (${FILE_SIZE} bytes)"
echo "  Output: $OUTPUT_FILE"

# Step 4: Test generate (optional, requires LLM key)
echo "[4/4] Testing generate pipeline (optional)..."
GENERATE_BODY='{"prompt": "Create a 2-slide overview of cloud computing benefits", "options": {"max_slides": 2}}'
GEN_CODE=$(curl -s -o /dev/null -w "%{http_code}" \
  -X POST "$API_URL/v1/generate" \
  -H "Authorization: Bearer $API_KEY" \
  -H "Content-Type: application/json" \
  -d "$GENERATE_BODY" \
  --max-time 30 2>/dev/null || echo "000")

if [[ "$GEN_CODE" == "200" || "$GEN_CODE" == "202" ]]; then
  echo "  OK: Generate endpoint responding ($GEN_CODE)"
elif [[ "$GEN_CODE" == "422" || "$GEN_CODE" == "500" ]]; then
  echo "  SKIP: Generate returned $GEN_CODE (likely no LLM key configured -- OK for smoke test)"
else
  echo "  SKIP: Generate returned $GEN_CODE (non-critical)"
fi

echo ""
echo "=== Smoke Test Passed ==="
echo "Render pipeline verified. PPTX output: $OUTPUT_FILE"
```

```python
#!/usr/bin/env python3
"""Validate all 5 demo deck IRs render to valid .pptx files.

Loads each demo IR from demos/*/ir.json, validates against the Pydantic
IR schema, renders via the render pipeline, and verifies the output.

Can run in two modes:
  1. API mode (default): POSTs to running API server
  2. Direct mode (--direct): Imports renderer directly (no server needed)

Usage:
  python scripts/validate-demos.py                    # API mode
  python scripts/validate-demos.py --direct            # Direct render
  python scripts/validate-demos.py --api-key dk_test_  # Custom key
"""

import argparse
import json
import os
import sys
from pathlib import Path

DEMOS_DIR = Path(__file__).parent.parent / "demos"
OUTPUT_DIR = Path(__file__).parent.parent / "tmp" / "demo-validation"
EXPECTED_DEMOS = [
    "mckinsey-strategy",
    "pe-deal-memo",
    "startup-pitch",
    "board-update",
    "product-launch",
]


def validate_pptx(filepath: Path) -> tuple[bool, str]:
    """Check file is valid PPTX (ZIP with PK header, >5KB)."""
    if not filepath.exists():
        return False, "File does not exist"
    size = filepath.stat().st_size
    if size < 5000:
        return False, f"Too small: {size} bytes (expected >5000)"
    with open(filepath, "rb") as f:
        header = f.read(2)
    if header != b"PK":
        return False, f"Invalid header: {header!r} (expected PK)"
    return True, f"Valid PPTX ({size:,} bytes)"


def validate_ir_schema(ir_path: Path) -> tuple[bool, str]:
    """Validate IR JSON against Pydantic schema."""
    try:
        sys.path.insert(0, str(Path(__file__).parent.parent))
        from src.deckforge.ir.presentation import Presentation
        with open(ir_path) as f:
            data = json.load(f)
        Presentation.model_validate(data)
        return True, "Schema valid"
    except Exception as e:
        return False, f"Schema error: {e}"


def render_via_api(ir_data: dict, output_path: Path, api_url: str, api_key: str) -> tuple[bool, str]:
    """Render IR via API POST."""
    import urllib.request
    import urllib.error

    body = json.dumps({"presentation": ir_data}).encode()
    req = urllib.request.Request(
        f"{api_url}/v1/render",
        data=body,
        headers={
            "Authorization": f"Bearer {api_key}",
            "Content-Type": "application/json",
        },
    )
    try:
        with urllib.request.urlopen(req, timeout=60) as resp:
            if resp.status != 200:
                return False, f"HTTP {resp.status}"
            output_path.parent.mkdir(parents=True, exist_ok=True)
            with open(output_path, "wb") as f:
                f.write(resp.read())
            return True, f"Rendered ({output_path})"
    except urllib.error.HTTPError as e:
        body_text = e.read().decode()[:200]
        return False, f"HTTP {e.code}: {body_text}"
    except Exception as e:
        return False, f"Error: {e}"


def render_direct(ir_data: dict, output_path: Path) -> tuple[bool, str]:
    """Render IR directly using Python renderer (no server needed)."""
    try:
        sys.path.insert(0, str(Path(__file__).parent.parent))
        from src.deckforge.ir.presentation import Presentation
        from src.deckforge.rendering.pptx_renderer import PptxRenderer
        from src.deckforge.layout.engine import LayoutEngine
        from src.deckforge.themes.registry import ThemeRegistry

        presentation = Presentation.model_validate(ir_data)
        theme = ThemeRegistry.get(presentation.theme or "corporate-blue")
        layout_engine = LayoutEngine()
        renderer = PptxRenderer()

        layout_results = layout_engine.layout(presentation, theme)
        pptx_bytes = renderer.render(presentation, layout_results, theme)

        output_path.parent.mkdir(parents=True, exist_ok=True)
        with open(output_path, "wb") as f:
            f.write(pptx_bytes)
        return True, f"Rendered ({output_path})"
    except Exception as e:
        return False, f"Render error: {e}"


def main():
    parser = argparse.ArgumentParser(description="Validate demo deck rendering")
    parser.add_argument("--direct", action="store_true", help="Use direct render (no API server)")
    parser.add_argument("--api-url", default="http://localhost:8000", help="API base URL")
    parser.add_argument("--api-key", default=os.environ.get("DECKFORGE_TEST_API_KEY", ""), help="API key")
    args = parser.parse_args()

    print("=== Demo Deck Validation ===")
    print(f"Mode: {'Direct' if args.direct else 'API'}")
    print(f"Demos: {DEMOS_DIR}")
    print()

    OUTPUT_DIR.mkdir(parents=True, exist_ok=True)
    results = []
    passed = 0
    failed = 0

    for demo_name in EXPECTED_DEMOS:
        ir_path = DEMOS_DIR / demo_name / "ir.json"
        output_path = OUTPUT_DIR / f"{demo_name}.pptx"

        print(f"[{demo_name}]")

        # Check IR exists
        if not ir_path.exists():
            print(f"  FAIL: {ir_path} not found")
            failed += 1
            continue

        # Load IR
        with open(ir_path) as f:
            ir_data = json.load(f)
        slide_count = len(ir_data.get("slides", []))
        print(f"  Loaded: {slide_count} slides")

        # Validate schema
        ok, msg = validate_ir_schema(ir_path)
        print(f"  Schema: {msg}")
        if not ok:
            failed += 1
            continue

        # Render
        if args.direct:
            ok, msg = render_direct(ir_data, output_path)
        else:
            if not args.api_key:
                print("  SKIP: No API key (set DECKFORGE_TEST_API_KEY or use --direct)")
                continue
            ok, msg = render_via_api(ir_data, output_path, args.api_url, args.api_key)
        print(f"  Render: {msg}")
        if not ok:
            failed += 1
            continue

        # Validate output
        ok, msg = validate_pptx(output_path)
        print(f"  Output: {msg}")
        if ok:
            passed += 1
        else:
            failed += 1

        print()

    print(f"=== Results: {passed}/{len(EXPECTED_DEMOS)} passed, {failed} failed ===")
    sys.exit(0 if failed == 0 else 1)


if __name__ == "__main__":
    main()
```
