class FinancialFormatter:
    @staticmethod
    def currency(value, symbol="$", precision=1, compact=True, negative_parens=True) -> str: ...
    @staticmethod
    def percentage(value, precision=1, is_decimal=True) -> str: ...
    @staticmethod
    def multiple(value, precision=1) -> str: ...
    @staticmethod
    def auto_format(value, format_type) -> str: ...

class ConditionalFormatter:
    @staticmethod
    def pos_neg_color(value, theme) -> str: ...
    @staticmethod
    def median_highlight(value, median, theme) -> str: ...
    @staticmethod
    def heatmap_gradient(value, min_val, max_val, theme) -> str: ...
    @staticmethod
    def traffic_light(status: str) -> str: ...

       - If `ir_slide.sort_by` is set, sorts rows by that column before rendering.
       - Applies conditional formatting: values above median get lightened positive bg, below get lightened negative bg (using `ConditionalFormatter.median_highlight()`).

       FINANCE_SLIDE_RENDERERS: dict[str, BaseFinanceSlideRenderer] = {

           "dcf_summary": DcfSummaryRenderer(),
           "waterfall_chart": WaterfallSlideRenderer(),

           return False
